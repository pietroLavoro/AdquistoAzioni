package com.tuorg.acquisti.repository;

import com.tuorg.acquisti.domain.Titolo;
import org.springframework.data.jpa.repository.JpaRepository;

import java.time.LocalDate;
import java.util.List;
import java.util.Optional;

public interface TitoloRepository extends JpaRepository<Titolo, Long> {

    // Lista todos los títulos ordenados por código (A→Z)
    List<Titolo> findAllByOrderByCodiceAsc();

    // Títulos emitidos hasta una fecha dada (inclusive), ordenados por código
    List<Titolo> findByDataEmissioneLessThanEqualOrderByCodiceAsc(LocalDate data);

    // Buscar por código exacto
    Optional<Titolo> findByCodice(String codice);
}
