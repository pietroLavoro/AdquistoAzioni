package com.tuorg.acquisti.domain;

import jakarta.persistence.*;
import java.math.BigDecimal;
import java.time.LocalDate;

@Entity
@Table(name = "movimento_titolo")
public class MovimentoTitolo {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    // T√çTULO comprado
    @ManyToOne(optional = false, fetch = FetchType.LAZY)
    @JoinColumn(name = "id_titolo", nullable = false)
    private Titolo titolo;

    // Fecha de la compra
    @Column(name = "data_investimento", nullable = false)
    private LocalDate dataInvestimento;

    // Cantidad total comprada
    @Column(name = "quantita", nullable = false)
    private Integer quantita;

    // Importe total
    @Column(name = "importo_acquisto", nullable = false, precision = 15, scale = 2)
    private BigDecimal importoAcquisto;

    // ---- GETTERS/SETTERS que usa el service ----
    public Long getId() { return id; }

    public Titolo getTitolo() { return titolo; }
    public void setTitolo(Titolo titolo) { this.titolo = titolo; }

    public LocalDate getDataInvestimento() { return dataInvestimento; }
    public void setDataInvestimento(LocalDate dataInvestimento) { this.dataInvestimento = dataInvestimento; }

    public Integer getQuantita() { return quantita; }
    public void setQuantita(Integer quantita) { this.quantita = quantita; }

    public BigDecimal getImportoAcquisto() { return importoAcquisto; }
    public void setImportoAcquisto(BigDecimal importoAcquisto) { this.importoAcquisto = importoAcquisto; }
}
