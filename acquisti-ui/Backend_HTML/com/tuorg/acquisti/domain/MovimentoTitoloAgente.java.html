package com.tuorg.acquisti.domain;

import jakarta.persistence.*;
import java.math.BigDecimal;

@Entity
@Table(name = "movimento_titolo_agente")
public class MovimentoTitoloAgente {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    // Cabecera (movimiento_titolo)
    @ManyToOne(optional = false, fetch = FetchType.LAZY)
    @JoinColumn(name = "id_movimento_titolo", nullable = false)
    private MovimentoTitolo movimentoTitolo;

    // Agente que participa
    @ManyToOne(optional = false, fetch = FetchType.LAZY)
    @JoinColumn(name = "id_agente", nullable = false)
    private Agente agente;

    // Cantidad asignada al agente
    @Column(name = "quantita", nullable = false)
    private Integer quantita;

    // Importe asignado al agente
    @Column(name = "importo_acquisto", nullable = false, precision = 15, scale = 2)
    private BigDecimal importoAcquisto;

    // ---- GETTERS/SETTERS que usa el service ----
    public Long getId() { return id; }

    public MovimentoTitolo getMovimentoTitolo() { return movimentoTitolo; }
    public void setMovimentoTitolo(MovimentoTitolo movimentoTitolo) { this.movimentoTitolo = movimentoTitolo; }

    public Agente getAgente() { return agente; }
    public void setAgente(Agente agente) { this.agente = agente; }

    public Integer getQuantita() { return quantita; }
    public void setQuantita(Integer quantita) { this.quantita = quantita; }

    public BigDecimal getImportoAcquisto() { return importoAcquisto; }
    public void setImportoAcquisto(BigDecimal importoAcquisto) { this.importoAcquisto = importoAcquisto; }
}
