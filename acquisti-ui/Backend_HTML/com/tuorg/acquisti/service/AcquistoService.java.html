package com.tuorg.acquisti.service;

import com.tuorg.acquisti.api.PreviewResponse;
import com.tuorg.acquisti.domain.Agente;
import com.tuorg.acquisti.domain.Movimento;
import com.tuorg.acquisti.domain.MovimentoTitolo;
import com.tuorg.acquisti.domain.MovimentoTitoloAgente;
import com.tuorg.acquisti.domain.Titolo;
import com.tuorg.acquisti.repository.AgenteRepository;
import com.tuorg.acquisti.repository.MovimentoRepository;
import com.tuorg.acquisti.repository.MovimentoTitoloAgenteRepository;
import com.tuorg.acquisti.repository.MovimentoTitoloRepository;
import com.tuorg.acquisti.repository.TitoloRepository;
import org.springframework.stereotype.Service;

import java.math.BigDecimal;
import java.time.LocalDate;

@Service
public class AcquistoService {

    private final RipartoService ripartoService;
    private final TitoloRepository titoloRepo;
    private final AgenteRepository agenteRepo;
    private final MovimentoTitoloRepository mtRepo;
    private final MovimentoTitoloAgenteRepository mtaRepo;
    private final MovimentoRepository movimentoRepo;

    public AcquistoService(RipartoService ripartoService,
                           TitoloRepository titoloRepo,
                           AgenteRepository agenteRepo,
                           MovimentoTitoloRepository mtRepo,
                           MovimentoTitoloAgenteRepository mtaRepo,
                           MovimentoRepository movimentoRepo) {
        this.ripartoService = ripartoService;
        this.titoloRepo = titoloRepo;
        this.agenteRepo = agenteRepo;
        this.mtRepo = mtRepo;
        this.mtaRepo = mtaRepo;
        this.movimentoRepo = movimentoRepo;
    }

    /** Confirma la compra: guarda cabecera, líneas por agente y (opcional) movimiento contable. */
    public void conferma(String codiceTitolo, LocalDate data, BigDecimal importoTot, int quantitaTot) {

        // 1) Recalcular preview para asegurar consistencia
        PreviewResponse prev = ripartoService.preview(codiceTitolo, data, importoTot, quantitaTot);

        // 2) Título (para la cabecera)
        Titolo titolo = titoloRepo.findByCodice(codiceTitolo)
                .orElseThrow(() -> new IllegalArgumentException("TITOLO_NON_TROVATO"));

        // 3) Cabecera MovimientoTitolo
        MovimentoTitolo mt = new MovimentoTitolo();
        mt.setTitolo(titolo);
        mt.setDataInvestimento(data);
        mt.setQuantita(quantitaTot);
        mt.setImportoAcquisto(importoTot);
        mt = mtRepo.save(mt);

        // 4) Líneas por agente (+ opcional: cargo contable negativo)
        for (PreviewResponse.RigaDTO r : prev.riparto()) {
            Agente a = agenteRepo.findById(r.agenteId())
                    .orElseThrow(() -> new IllegalArgumentException("AGENTE_NON_TROVATO"));

            MovimentoTitoloAgente mta = new MovimentoTitoloAgente();
            mta.setMovimentoTitolo(mt);
            mta.setAgente(a);
            mta.setQuantita(r.quantitaAgente());
            mta.setImportoAcquisto(r.importoAgente());
            mtaRepo.save(mta);

            // --- OPCIONAL: crear Movimiento contable (negativo) por el importe del agente ---
            if (r.importoAgente() != null && r.importoAgente().signum() > 0) {
                Movimento mov = new Movimento();
                mov.setAgente(a);
                mov.setData(data);
                // cargo (salida de dinero) como negativo
                mov.setImporto(r.importoAgente().negate());
                movimentoRepo.save(mov);
            }
        }
    }
}
