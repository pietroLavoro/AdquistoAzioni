package com.tuorg.acquisti.service;

import com.tuorg.acquisti.domain.Agente;
import com.tuorg.acquisti.repository.AgenteRepository;
import com.tuorg.acquisti.repository.MovimentoRepository;
import org.springframework.stereotype.Service;

import java.math.BigDecimal;
import java.time.LocalDate;
import java.util.LinkedHashMap;
import java.util.List;
import java.util.Map;

@Service
public class DominioService {

    private final AgenteRepository agenteRepo;
    private final MovimentoRepository movimentoRepo;

    public DominioService(AgenteRepository agenteRepo, MovimentoRepository movimentoRepo) {
        this.agenteRepo = agenteRepo;
        this.movimentoRepo = movimentoRepo;
    }

    /** Devuelve los agentes activos a la fecha dada. */
    public List<Agente> agentesAttiviAlla(LocalDate data) {
        return agenteRepo.findAttiviAlla(data);           // <--- nombre correcto
    }

    /** Saldo de un agente hasta la fecha (incluida). */
    public BigDecimal saldoAgenteFinoA(Long agenteId, LocalDate finoA) {
        return movimentoRepo.saldoAgenteFinoA(agenteId, finoA);
    }

    /** Mapa agenteId -> saldo (para todos los activos a la fecha). */
    public Map<Long, BigDecimal> saldiAgentiAttiviAlla(LocalDate data) {
        List<Agente> attivi = agenteRepo.findAttiviAlla(data);   // <--- nombre correcto
        Map<Long, BigDecimal> saldi = new LinkedHashMap<>();
        for (Agente a : attivi) {
            BigDecimal saldo = movimentoRepo.saldoAgenteFinoA(a.getId(), data);
            saldi.put(a.getId(), saldo);
        }
        return saldi;
    }
}
