package com.tuorg.acquisti.service;

import com.tuorg.acquisti.api.AcquistoDettaglioResponse;
import com.tuorg.acquisti.api.AcquistoSummaryResponse;
import com.tuorg.acquisti.domain.MovimentoTitolo;
import com.tuorg.acquisti.domain.MovimentoTitoloAgente;
import com.tuorg.acquisti.repository.MovimentoTitoloAgenteRepository;
import com.tuorg.acquisti.repository.MovimentoTitoloRepository;
import org.springframework.stereotype.Service;

import java.util.List;

@Service
public class ReportingService {

    private final MovimentoTitoloRepository mtRepo;
    private final MovimentoTitoloAgenteRepository mtaRepo;

    public ReportingService(MovimentoTitoloRepository mtRepo,
                            MovimentoTitoloAgenteRepository mtaRepo) {
        this.mtRepo = mtRepo;
        this.mtaRepo = mtaRepo;
    }

    /** Lista simple de compras (cabeceras). */
    public List<AcquistoSummaryResponse> listAcquisti() {
        return mtRepo.findAll().stream()
                .map(this::toSummary)
                .toList();
    }

    /** Detalle de una compra: cabecera + l√≠neas por agente. */
    public AcquistoDettaglioResponse dettaglio(Long movimentoTitoloId) {
        MovimentoTitolo mt = mtRepo.findById(movimentoTitoloId)
                .orElseThrow(() -> new IllegalArgumentException("MOVIMENTO_TITOLO_NON_TROVATO"));

        var righe = mtaRepo.findByMovimentoTitolo_Id(movimentoTitoloId).stream()
                .map(this::toRiga)
                .toList();

        return new AcquistoDettaglioResponse(
                mt.getId(),
                mt.getTitolo().getCodice(),
                mt.getTitolo().getDescrizione(),
                mt.getDataInvestimento(),
                mt.getQuantita(),
                mt.getImportoAcquisto(),
                righe
        );
    }

    // ---- mapeos ----
    private AcquistoSummaryResponse toSummary(MovimentoTitolo mt) {
        return new AcquistoSummaryResponse(
                mt.getId(),
                mt.getTitolo().getCodice(),
                mt.getTitolo().getDescrizione(),
                mt.getDataInvestimento(),
                mt.getQuantita(),
                mt.getImportoAcquisto()
        );
    }

    private AcquistoDettaglioResponse.Riga toRiga(MovimentoTitoloAgente r) {
        return new AcquistoDettaglioResponse.Riga(
                r.getAgente().getId(),
                r.getAgente().getCodiceFiscale(),
                r.getQuantita(),
                r.getImportoAcquisto()
        );
    }
}
