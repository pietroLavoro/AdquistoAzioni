package com.tuorg.acquisti.service;

import com.tuorg.acquisti.api.PreviewResponse;
import com.tuorg.acquisti.domain.Agente;
import com.tuorg.acquisti.domain.Titolo;
import com.tuorg.acquisti.repository.AgenteRepository;
import com.tuorg.acquisti.repository.MovimentoRepository;
import com.tuorg.acquisti.repository.TitoloRepository;
import org.springframework.stereotype.Service;

import java.math.BigDecimal;
import java.math.RoundingMode;
import java.time.LocalDate;
import java.util.ArrayList;
import java.util.Comparator;
import java.util.List;

/**
 * Calcula la pre-visualización del reparto de una compra.
 * Reparte proporcionalmente al saldo de cada agente a la fecha.
 */
@Service
public class RipartoService {

    private final AgenteRepository agenteRepo;
    private final MovimentoRepository movimentoRepo;
    private final TitoloRepository titoloRepo;

    public RipartoService(AgenteRepository agenteRepo,
                          MovimentoRepository movimentoRepo,
                          TitoloRepository titoloRepo) {
        this.agenteRepo = agenteRepo;
        this.movimentoRepo = movimentoRepo;
        this.titoloRepo = titoloRepo;
    }

    /**
     * Calcula el reparto previo.
     *
     * @param codiceTitolo código del título
     * @param data         fecha de compra
     * @param importoTot   importe total (positivo)
     * @param quantitaTot  cantidad total (entero >= 1)
     * @return PreviewResponse con el desglose por agente
     */
    public PreviewResponse preview(String codiceTitolo,
                                   LocalDate data,
                                   BigDecimal importoTot,
                                   int quantitaTot) {

        // ------- Validaciones básicas -------
        if (quantitaTot <= 0) {
            throw new IllegalArgumentException("QUANTITA_TOTALE_INVALIDA");
        }
        if (importoTot == null || importoTot.compareTo(BigDecimal.ZERO) <= 0) {
            throw new IllegalArgumentException("IMPORTO_TOTALE_INVALIDO");
        }

        // Título existente y fecha válida (compra >= emisión)
        Titolo titolo = titoloRepo.findByCodice(codiceTitolo)
                .orElseThrow(() -> new IllegalArgumentException("TITOLO_NON_TROVATO"));
        if (titolo.getDataEmissione() != null && data.isBefore(titolo.getDataEmissione())) {
            throw new IllegalArgumentException("DATA_COMPRA_PRE_EMISSIONE");
        }

        // ------- Participantes: agentes activos a la fecha + saldos -------
        List<Agente> attivi = agenteRepo.findAttiviAlla(data);
        if (attivi.isEmpty()) {
            throw new IllegalArgumentException("NESSUN_AGENTE_PARTECIPANTE");
        }

        // Cargamos saldos y nos quedamos solo con saldos > 0
        List<RigaCalcolo> righe = new ArrayList<>();
        BigDecimal saldoTotale = BigDecimal.ZERO;

        for (Agente a : attivi) {
            BigDecimal saldo = movimentoRepo.saldoAgenteFinoA(a.getId(), data);
            if (saldo != null && saldo.compareTo(BigDecimal.ZERO) > 0) {
                righe.add(new RigaCalcolo(a.getId(), a.getCodiceFiscale(), saldo));
                saldoTotale = saldoTotale.add(saldo);
            }
        }

        if (righe.isEmpty() || saldoTotale.compareTo(BigDecimal.ZERO) <= 0) {
            throw new IllegalArgumentException("NESSUN_AGENTE_PARTECIPANTE");
        }

        // ------- Reparto por importe (2 decimales) y por cantidad (entera) -------
        // 1) Cantidad base = floor(quantitaTot * saldo / saldoTotale)
        int asignadas = 0;
        for (RigaCalcolo r : righe) {
            BigDecimal quota = r.saldo.multiply(BigDecimal.valueOf(quantitaTot))
                    .divide(saldoTotale, 12, RoundingMode.HALF_UP); // alta precisión intermedia
            int base = quota.setScale(0, RoundingMode.FLOOR).intValue();
            r.quantitaBase = base;
            r.restoQuantita = quota.subtract(BigDecimal.valueOf(base)); // parte fraccionaria
            asignadas += base;

            // Importe parcial proporcional (2 decimales)
            r.importo = importoTot.multiply(r.saldo)
                    .divide(saldoTotale, 2, RoundingMode.HALF_UP);
        }

        // 2) Reparto de sobrantes por mayor resto
        int sobrantes = quantitaTot - asignadas;
        if (sobrantes > 0) {
            righe.sort(Comparator.comparing((RigaCalcolo rc) -> rc.restoQuantita).reversed());
            for (int i = 0; i < sobrantes; i++) {
                righe.get(i % righe.size()).quantitaBase += 1;
            }
        }

        // ------- Construir DTO de respuesta -------
        List<PreviewResponse.RigaDTO> riparto = new ArrayList<>();
        for (RigaCalcolo r : righe) {
            riparto.add(new PreviewResponse.RigaDTO(
                    r.agenteId,
                    r.codiceFiscale,
                    r.importo,
                    r.quantitaBase
            )); // <<---- si tu DTO de fila tiene otro nombre/orden, ajusta aquí
        }

        return new PreviewResponse(
                codiceTitolo,
                data,             // <<---- si tu PreviewResponse espera String (yyyy-MM-dd), formatea aquí
                importoTot,
                quantitaTot,
                riparto           // <<---- lista de filas
        );
    }

    /**
     * Estructura interna para cálculos intermedios.
     */
    private static class RigaCalcolo {
        final Long agenteId;
        final String codiceFiscale;
        final BigDecimal saldo;
        BigDecimal importo = BigDecimal.ZERO;
        int quantitaBase = 0;
        BigDecimal restoQuantita = BigDecimal.ZERO;

        RigaCalcolo(Long agenteId, String codiceFiscale, BigDecimal saldo) {
            this.agenteId = agenteId;
            this.codiceFiscale = codiceFiscale;
            this.saldo = saldo;
        }
    }
}
