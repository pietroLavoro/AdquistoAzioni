package com.tuorg.acquisti.api;

import com.tuorg.acquisti.repository.TitoloRepository;
import org.springframework.web.bind.annotation.*;
import java.util.List;

@RestController
@RequestMapping("/api/titolo")
public class TitoloController {

    private final TitoloRepository titoloRepository;

    public TitoloController(TitoloRepository titoloRepository) {
        this.titoloRepository = titoloRepository;
    }

    @GetMapping
    public List<TitoloDTO> list() {
        return titoloRepository.findAllByOrderByCodiceAsc()
                .stream()
                .map(t -> new TitoloDTO(t.getCodice(), t.getDescrizione()))
                .toList();
    }
}
