package com.tuorg.acquisti.api;

import com.tuorg.acquisti.repository.TitoloRepository;
import com.tuorg.acquisti.service.AcquistoService;
import com.tuorg.acquisti.service.ReportingService;
import com.tuorg.acquisti.service.RipartoService;

import jakarta.validation.Valid;
import org.springframework.http.HttpStatus;
import org.springframework.web.bind.annotation.*;
import org.springframework.web.server.ResponseStatusException;

import java.time.LocalDate;
import java.time.format.DateTimeFormatter;
import java.util.List;

@RestController
@RequestMapping("/api/acquisto")
public class AcquistoController {

    private final RipartoService riparto;
    private final AcquistoService acquistoService;
    private final ReportingService reportingService;
    private final TitoloRepository titoloRepository;

    public AcquistoController(RipartoService riparto,
                              AcquistoService acquistoService,
                              ReportingService reportingService,
                              TitoloRepository titoloRepository) {
        this.riparto = riparto;
        this.acquistoService = acquistoService;
        this.reportingService = reportingService;
        this.titoloRepository = titoloRepository;
    }

    // ---------- PREVIEW ----------
    @PostMapping("/preview")
    public PreviewResponse preview(@Valid @RequestBody PreviewRequest req) {

        // 1) validar que el t√≠tulo existe
        titoloRepository.findByCodice(req.getTitoloCodice())
                .orElseThrow(() -> new ResponseStatusException(
                        HttpStatus.BAD_REQUEST,
                        "Titolo '" + req.getTitoloCodice() + "' non trovato"
                ));

        // 2) parsear fecha 'yyyy-MM-dd'
        LocalDate data = LocalDate.parse(req.getDataCompra(), DateTimeFormatter.ISO_LOCAL_DATE);

        // 3) validar importes/cantidades > 0 (si falla, 400 claro)
        if (req.getImportoTotale() == null || req.getImportoTotale().doubleValue() <= 0) {
            throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "importoTotale deve essere > 0");
        }
        if (req.getQuantitaTotale() == null || req.getQuantitaTotale() <= 0) {
            throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "quantitaTotale deve essere > 0");
        }

        // 4) delegar al servicio
        return riparto.preview(
                req.getTitoloCodice(),
                data,
                req.getImportoTotale(),
                req.getQuantitaTotale()
        );
    }

    // ---------- CONFERMA ----------
    @PostMapping("/conferma")
    public void conferma(@Valid @RequestBody PreviewRequest req) {
        titoloRepository.findByCodice(req.getTitoloCodice())
                .orElseThrow(() -> new ResponseStatusException(
                        HttpStatus.BAD_REQUEST,
                        "Titolo '" + req.getTitoloCodice() + "' non trovato"
                ));

        LocalDate data = LocalDate.parse(req.getDataCompra(), DateTimeFormatter.ISO_LOCAL_DATE);

        acquistoService.conferma(
                req.getTitoloCodice(),
                data,
                req.getImportoTotale(),
                req.getQuantitaTotale()
        );
    }


    // ---------- LISTA ----------
    @GetMapping
    public List<AcquistoSummaryResponse> list() {
        return reportingService.listAcquisti();
    }

    // ---------- DETTAGLIO ----------
    @GetMapping("/{id}")
    public AcquistoDettaglioResponse dettaglio(@PathVariable Long id) {
        return reportingService.dettaglio(id);
    }
}
