package com.tuorg.acquisti.api;

import com.tuorg.acquisti.service.AnalisiService;
import com.tuorg.acquisti.api.AgenteSaldoDTO;
import com.tuorg.acquisti.api.AgentiAttiviDTO;
import com.tuorg.acquisti.api.SuggerimentoDataDTO;
import org.springframework.format.annotation.DateTimeFormat;
import org.springframework.http.HttpStatus;
import org.springframework.web.bind.annotation.*;
import org.springframework.web.server.ResponseStatusException;

import java.time.LocalDate;
import java.util.List;

@RestController
@RequestMapping("/api/analisi")
public class AnalisiController {

    private final AnalisiService analisi;

    public AnalisiController(AnalisiService analisi) {
        this.analisi = analisi;
    }

    /** Saldos de TODOS los agentes a la fecha (si no se pasa, hoy) */
    @GetMapping("/saldi")
    public List<AgenteSaldoDTO> saldiAllaData(
            @RequestParam(name = "data", required = false)
            @DateTimeFormat(iso = DateTimeFormat.ISO.DATE) LocalDate data) {

        LocalDate d = (data != null) ? data : LocalDate.now();
        return analisi.saldiTuttiAllaData(d);
    }

    /**
     * Sugerir una fecha de compra según el título y #agentes.
     * Acepta 'codiceTitolo' (o 'codice' por compatibilidad) y 'numAgenti' (>=1).
     * Devuelve la fecha en formato dd-MM-yyyy (la conversión la hace el service).
     */
    @GetMapping("/suggerisci-data")
    public SuggerimentoDataDTO suggerisciData(
            @RequestParam(name = "codiceTitolo", required = false) String codiceTitolo,
            @RequestParam(name = "codice", required = false) String codice,
            @RequestParam(name = "numAgenti", required = false) Integer numAgenti) {

        // Normaliza el código del título admitiendo ambos nombres de parámetro
        String codiceEffettivo = (codiceTitolo != null && !codiceTitolo.isBlank())
                ? codiceTitolo
                : (codice != null ? codice : null);

        if (codiceEffettivo == null || codiceEffettivo.isBlank()) {
            throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "codiceTitolo obbligatorio");
        }
        if (numAgenti == null || numAgenti < 1) {
            throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "numAgenti deve essere >= 1");
        }

        return analisi.suggerisciData(codiceEffettivo, numAgenti);
    }

    /** Reset de datos de prueba (borra compras/movimientos y recarga seed) */
    @PostMapping("/reset")
    public void reset() {
        analisi.resetTestData();
    }

    /** Agentes activos a la fecha (si no se pasa, hoy) */
    @GetMapping("/agenti-attivi")
    public AgentiAttiviDTO agentiAttiviAllaData(
            @RequestParam(name = "data", required = false)
            @DateTimeFormat(iso = DateTimeFormat.ISO.DATE) LocalDate data) {

        LocalDate d = (data != null) ? data : LocalDate.now();
        return analisi.agentiAttiviAllaData(d);
    }
}
