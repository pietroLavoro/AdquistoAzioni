package com.tuorg.acquisti.api;

import jakarta.validation.constraints.*;
import java.math.BigDecimal;

public record ConfirmRequest(
        @NotBlank String titoloCodice,
        @Pattern(regexp="^\\d{8}$") String dataCompra,  // yyyyMMdd
        @DecimalMin("0.01") BigDecimal importoTotale,
        @Min(1) int quantitaTotale
) {}
