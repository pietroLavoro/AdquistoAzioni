package com.tuorg.acquisti.api;

import jakarta.validation.constraints.*;

public class PreviewRequest {

        @NotBlank(message = "titoloCodice es obligatorio")
        private String titoloCodice;

        @Pattern(regexp = "\\d{4}-\\d{2}-\\d{2}", message = "dataCompra debe tener formato yyyy-MM-dd")
        private String dataCompra;

        @DecimalMin(value = "0.01", message = "importoTotale debe ser > 0")
        private java.math.BigDecimal importoTotale;

        @Min(value = 1, message = "quantitaTotale debe ser >= 1")
        private Integer quantitaTotale;

        // getters/setters
        public String getTitoloCodice() { return titoloCodice; }
        public void setTitoloCodice(String titoloCodice) { this.titoloCodice = titoloCodice; }

        public String getDataCompra() { return dataCompra; }
        public void setDataCompra(String dataCompra) { this.dataCompra = dataCompra; }

        public java.math.BigDecimal getImportoTotale() { return importoTotale; }
        public void setImportoTotale(java.math.BigDecimal importoTotale) { this.importoTotale = importoTotale; }

        public Integer getQuantitaTotale() { return quantitaTotale; }
        public void setQuantitaTotale(Integer quantitaTotale) { this.quantitaTotale = quantitaTotale; }
}
