package com.tuorg.acquisti.api;

import java.math.BigDecimal;
import java.time.LocalDate;
import java.util.List;

/** Detalle de una compra con desglose por agente. */
public record AcquistoDettaglioResponse(
        Long id,
        String titoloCodice,
        String titoloDescrizione,
        LocalDate dataCompra,
        int quantitaTotale,
        BigDecimal importoTotale,
        List<Riga> riparto
) {
    public record Riga(
            Long agenteId,
            String codiceFiscale,
            Integer quantitaAgente,
            BigDecimal importoAgente
    ) {}
}
