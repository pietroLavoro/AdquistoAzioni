package com.tuorg.acquisti;

import com.tuorg.acquisti.service.DominioService;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;

import java.time.LocalDate;
import java.util.Map;

@RestController
public class HealthController {

    private final DominioService dominio;

    public HealthController(DominioService dominio) {
        this.dominio = dominio;
    }

    @GetMapping("/api/check/saldi")
    public Map<Long, ?> saldi(@RequestParam String data) {
        LocalDate d = LocalDate.parse(data); // por ahora ISO yyyy-MM-dd para probar
        return dominio.saldiAgentiAttiviAlla(d);
    }
}
